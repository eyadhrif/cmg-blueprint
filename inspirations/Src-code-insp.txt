        var el = document.querySelector('.bc-price-main');
        if (!el) return null;
        var n = parseFloat((el.textContent || '').replace(/[^\d.]/g, ''));
        return isFinite(n) && n > 0 ? n : null;
      } catch (_) { return null; }
    }

    // Maps the price actually rendered on the PDP to the marketing label
    // we use everywhere ($179 → p179 etc). Intelligems' own group names
    // (set by the merchant in Intelligems UI) are surfaced separately as
    // `ig_group_name` — keep this label as the canonical price-bucket id
    // for Meta breakdowns regardless of Intelligems naming.
    function priceToGroupLabel(price) {
      if (price === 179) return 'p179';
      if (price === 199) return 'p199';
      if (price === 219) return 'p219';
      if (price === 229) return 'p229';
      if (price === 249) return 'p249';
      return null;
    }

        for (var i = 0; i < experiments.length; i++) {
          var exp = experiments[i];
          var tg = null;

              experiment_name: exp.name,
